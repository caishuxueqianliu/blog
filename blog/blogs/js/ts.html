  var xhr = new XMLHttpRequest()

    # 类的泛型

    // 类的泛型
class minClass<T> {
  public list:T[]=[]
  add(value:T):void{
    this.list.push(value)
  }
  min():T{
    let minnum:T = this.list[0]
    for(let i=0;i<this.list.length;i++){
      if(this.list[i]<minnum){
        minnum = this.list[i]
      }
    }
    return minnum
  }
}
let xx = new minClass<number>()
xx.add(2)
xx.add(23)
xx.add(1)
console.log(xx.min())
let yy = new minClass<string>()
yy.add('gg')
yy.add('c')
yy.add('e')
console.log(yy.min())
